sysstat 搜集和报告系统活动

sar命令用来搜集、报告和储存系统活动信息。 

sar -o sar_`hostname`_`date +'%Y%m%d_%H%M%S'` 10 8640 >/dev/null 2>&1 &

# MySQL性能分析
./tuning-primer.sh all >tuning-primer_`date +'%Y%m%d_%H%M%S'`.txt
mysql -uroot -pxxxxx -e 'show processlist;'

# CPU:
sar -u all -f datafile
# 网络：
sar -n DEV -f datafile|grep -E "eth0|IFACE"
# 内存：
sar -r -f  datafile
# 显示最近24小时网络计数器
sar -n DEV -f /var/log/sa/sa24 | more
  • 用sar显示实时情况 
    sar 4 5
