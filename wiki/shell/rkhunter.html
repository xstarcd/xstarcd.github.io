rkhunter

  • http://sourceforge.net/projects/rkhunter/
  • http://www.rootkit.nl/projects/ 
# Ubuntu下安装
apt-get install rkhunter

# 下载
cd /tmp
wget http://ncu.dl.sourceforge.net/project/rkhunter/rkhunter/1.4.0/rkhunter-1.4.0.tar.gz

# 安装
tar -xvf rkhunter-1.4.0.tar.gz
cd rkhunter-1.4.0
./installer.sh --layout default --install

# 更新
/usr/local/bin/rkhunter --update
/usr/local/bin/rkhunter --propupd

# Setting Cronjob and Email Alerts
cat > /etc/cron.daily/rkhunter.sh <<EOF
#!/bin/sh
(
/usr/local/bin/rkhunter --versioncheck
/usr/local/bin/rkhunter --update
/usr/local/bin/rkhunter --cronjob --report-warnings-only
) | /bin/mail -s 'rkhunter Daily Run (PutYourServerNameHere)' your@email.com
EOF
chmod 755 /etc/cron.daily/rkhunter.sh

# Manual Scan and Usage
rkhunter --check
rkhunter --help
