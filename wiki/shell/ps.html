ps

# 查看进程准确启动时间
ps -eo pid,tty,user,comm,stime,etime | grep -E "PID|mysql"
# ELAPSED: day-HH:MM:SS，用当前时间减去ELAPSED，即为启动时间。
ls -l /proc/`ps aux|grep java|grep -v grep |awk '{print $2}'`/cmdline

ps -AlF   #显示进程结构
ps -AlFH  #在进程后显示线程

# 只显示Lighttpd的进程ID
ps -C lighttpd -o pid=
pgrep lighttpd
pgrep -u vivek php-cgi

ps -auxf | sort -nr -k 4 | head -10 #找出消耗内存最多的前10名进程
ps -auxf | sort -nr -k 3 | head -10 #找出使用CPU最多的前10名进程
