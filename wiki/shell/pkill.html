# 强制踢人命令格式：pkill -kill -t tty
# tty　所踢用户的TTY
# 如上踢出tty2上的用户的命令为： pkill -kill -t tty2
# 踢除当前所有登录，只保留当前使用的tty/pts

Me="`tty|sed 's#/dev/##g'`";w|awk '{if($2~"^pts/|^tty")print $2}'|grep -v "$Me"|xargs -I {} pkill -kill -t {};w
