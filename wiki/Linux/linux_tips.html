last与lastb(wtmp,btmp)

# 清空命令：
cat /dev/null > /var/log/wtmp
cat /dev/null > /var/log/btmp
cat /dev/null > /var/log/lastlog

# 查看命令：
lastlog  #/var/log/lastlog - reports the most recent login of all users or of a given user
last     #/var/log/wtmp - last, lastb - show listing of last logged in users
lastb    #/var/log/btmp
