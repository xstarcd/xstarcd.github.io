Ubuntu_tips

  • Ubuntu安装时,software select中"Basic Ubuntu server"、"Virtual machine host"包括哪些包 
    #tasksel - Debian 系统安装任务选择工具
#tasksel-data - 用于安装 Debian 系统的官方系统安装任务
sudo apt-get install tasksel

# 列出安装任务名称
tasksel --list-tasks|grep 'Basic Ubuntu server'
i server    Basic Ubuntu server

# 查看安装任务包内容
tasksel --task-packages server

# 不带参数执行tasksel，进入选择安装界面
    • 用法 
      tasksel install <软件集>
tasksel remove <软件集>
tasksel [选项]
    -t, --test          测试模式，不会真正执行任何操作
        --new-install   自动安装某些软件集
        --list-tasks    列出将要显示的软件集并退出
        --task-packages 列出某软件集中的软件包
        --task-desc     显示某软件集的说明信息
  • Ubuntu重新生成OpenSSH Host Keys 
    rm /etc/ssh/ssh_host_*
dpkg-reconfigure -fnoninteractive -pcritical openssh-server
  • lightdm开启XDMCP 
    # /etc/lightdm/lightdm.conf
[XDMCPServer]
enabled=true

# 增加登录选项，允许输入用户名
greeter-show-manual-login=true
