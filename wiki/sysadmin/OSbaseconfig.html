系统基本设置

主机名修改

# CentOS
h="TMXLlvs2" \
&& hostname $h \
&& sed -i.bak "s/\(HOSTNAME=\).*$/\1$h/g" /etc/sysconfig/network \
&& grep "127.0.0.1.*$h" /etc/hosts || sed -i "s/\(127.0.0.1.*$\)/\1 $h/g" /etc/hosts \
&& ping -c 4 $h


# Ubuntu
h="TMXLlvs2" \
&& hostname $h \
&& echo "$h" > /etc/hostname \
&& grep "127.0.0.1.*$h" /etc/hosts || sed -i "s/\(127.0.0.1.*$\)/\1 $h/g" /etc/hosts \
&& ping -c 4 $h
